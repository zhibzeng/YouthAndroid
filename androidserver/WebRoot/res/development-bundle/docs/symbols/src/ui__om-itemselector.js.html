240             this.leftPanel = $('<table cellspacing="0" cellpadding="0"></table>').append(thead)

246         _init:function(){
247             var op=this.options,
248                 dataSource=op.dataSource;
249             this.leftPanel.find(".om-itemselector-title").html($.om.lang._get(op,"omItemSelector","availableTitle"));
250             this.rightPanel.find(".om-itemselector-title").html($.om.lang._get(op,"omItemSelector","selectedTitle"));
251             this.element.css({width:op.width,height:op.height});
252             this._buildToolbar();
253             this._resizePanel();//调整左右fieldset大小
254             this._bindEvents();

256                 var self=this;
257                 $.ajax({
258                     url: dataSource,
259                     method: 'GET',
260                     dataType: 'json',
261                     success: function(data, textStatus){
262                         if (self._trigger("onSuccess",null,data,textStatus) === false) {
263                             return;
264                         }
265                         data=op.preProcess(data);
266                         op.dataSource=data;
267                         self._buildList();
268                     },
269                     error: function(XMLHttpRequest, textStatus, errorThrown){
270                     	self._trigger("onError",null,XMLHttpRequest, textStatus, errorThrown);
271                     }
272                 });
273             }else{

275             }
276              
277             this._refreshPageButton(this.leftPanel);
278             this._refreshPageButton(this.rightPanel);

293                 rightItemsContainer=$('.om-itemselector-items',rp),

301         _buildList:function(){
302             var op=this.options;
303                 dataSource = op.dataSource,
304                 value = op.value,
305                 fmt=op.clientFormatter,
306                 leftHtml='',
307                 rightHtml='',
308                 // {text:'abc',value:2}的value是否在value:[0,2,4]这样的数组中
309                 inArray=function(data,valueArr){
310                     for(var i=0,len=valueArr.length;i<len;i++){
311                         if(data.value===valueArr[i]){
312                             return true;
313                         }
314                     }
315                     return false;
316                 },
317                 buildHtml=fmt?function(index,data){
318                     return '<dt _index="'+index+'">'+'<span class="checkbox"/>'+fmt(data,index)+'</dt>';
319                 }:function(index,data){
320                     return '<dt _index="'+index+'">'+'<span class="checkbox"/>'+data.text+'</dt>';
321                 };
322             if($.isArray(dataSource) && jQuery.isArray(value)){
323                 $.each(dataSource,function(index,data){
324                     if(inArray(data,value)){//在value中，要放到右边
325                         rightHtml+=buildHtml(index,data);
326                     }else{//不在value中,放到左边
327                         leftHtml+=buildHtml(index,data);
328                     }
329                 });
330             }
331             $('.om-itemselector-items>dl',this.leftPanel).html(leftHtml);
332             $('.om-itemselector-items>dl',this.rightPanel).html(rightHtml);
333             var items = $('.om-itemselector-items'),itemdtH =$('>dl>dt',items).outerHeight(),
334             h =items.outerHeight(),offset = itemdtH - h%itemdtH;
335             items.outerHeight(h+offset);           
336         },

403         _page: function(btn, isup){
404         	var $items = isup?btn.next():btn.prev(),
405         	$dl = $items.children("dl"), h=$items.outerHeight()-2,dlH = $dl.outerHeight(),
406         	dltop =$dl.offset().top, nextbtn, top = this.element.data("dltop")+20;
407         	if(isup){
408         		$dl.offset({top: dltop+h});
409         		nextbtn = $items.next();
410         		if((dltop =$dl.offset().top) >0||($dl.outerHeight()- dltop-top) >h){
411         			nextbtn.removeAttr("disabled").removeClass("om-itemselector-down-disabled");
412         		}
413         		if(dltop > 0 && dltop < h){
414         			btn.attr("disabled","disabled").addClass("om-itemselector-up-disabled");
415         		}
416         	}else{
417         		$dl.offset({top: dltop-h});
418         		nextbtn = $items.prev();
419             	nextbtn.removeAttr("disabled").removeClass("om-itemselector-up-disabled");
420             	if((dltop =$dl.offset().top) <0 && (dlH + dltop - top)<=h){
421             		btn.attr("disabled","disabled").addClass("om-itemselector-down-disabled");
422             	}
423         	}
424         },
425         _refreshHeaderCheckbox: function(panel){
426         	var $dt = $('.om-itemselector-items>dl>dt', panel);
427         	$dt = $(this._getPageItems(panel, $dt));
428         	var selects = $dt.filter('.om-state-highlight').length;
429         	$('.header span.checkbox', panel).toggleClass("selected", selects>0 && $dt.length===selects);
430         },
431         _refreshPageButton: function(panel){
432         	var $items=$('.om-itemselector-items',panel), $dl=$('.om-itemselector-items >dl', panel),
433             $up = $('.om-itemselector-up', panel), $down = $('.om-itemselector-down', panel),
434             itemsH =$items.outerHeight(),dlH = $dl.outerHeight(),
435             dltop = $dl.offset().top, top = this.element.data("dltop")+20;
436         	if(dlH > 20 &&(top - dltop) >=dlH){
437         		$dl.offset({top: dltop+itemsH});
438         		dltop = $dl.offset().top;
439     		}
440         	if(dlH > itemsH){
441         		if($up.is(":hidden")){
442             		$items.outerHeight(itemsH - $up.outerHeight()*2);
443             		$up.show();
444             		$down.show();
445         		}
446         		if(dltop>0 && dltop < itemsH){
447             		$up.attr("disabled","disabled").addClass("om-itemselector-up-disabled");
448             	}else{
449             		$up.removeAttr("disabled").removeClass("om-itemselector-up-disabled");
450             	}
451         		if(dltop >0 ||(dlH + dltop- top) >(itemsH)){
452         			$down.removeAttr("disabled").removeClass("om-itemselector-down-disabled");
453         		}else{
454         			$down.attr("disabled","disabled").addClass("om-itemselector-down-disabled");
455         		}
456         	}else{
457         		if($up.is(":visible")){
458         			$items.outerHeight(itemsH + $up.outerHeight()*2);
459         			$up.hide();
460         			$down.hide();
461         		}
462         	}
463         	
464         },
465         _getPageItems:function(panel, $dt){
466         	var items = $(".om-itemselector-items", panel), itemsH = items.outerHeight(),
467         	$dl = $(">dl", items), pageItems =[], hasPageButton = items.next(":visible").length>0,
468         	dtH = $dt.outerHeight(), num = itemsH/dtH, dltop = $dl.offset().top,
469         	top = this.element.data("dltop");
470             dltop = hasPageButton ? top - dltop+20 : top-dltop;
471         		pageItems = $.grep($dt, function(n, i){
472         			return i < num*(dltop/itemsH + 1)-1 && i >= num*(dltop/itemsH);
473         		});
474         	return pageItems;
475         },

485         _moveItemsToTarget:function(selector,isLeftToRight){
486             var self = this, fromPanel=isLeftToRight?this.leftPanel:this.rightPanel,
487                 selectedItems=$('.om-itemselector-items>dl>dt'+selector,fromPanel);
488             if(selectedItems.size()==0)
489                 return;
490             var toPanel=isLeftToRight?this.rightPanel:this.leftPanel,
491                 op=this.options,
492                 itemData=[];
493             selectedItems.each(function(){
494                 itemData.push(op.dataSource[$(this).attr('_index')]);
495             });
496             //先触发onBeforeItemSelect或onBeforeItemDeselect事件，然后移动并触发onItemSelect或onItemDeselect事件
497             if(isLeftToRight){
498                 if(self._trigger("onBeforeItemSelect",null,itemData)===false){
